Create a detailed view component for a single seat race session.

Props:
- `sessionId: string` - Session ID to display
- `onClose: () => void` - Close callback (for slide-out panel)
- `onDelete?: () => void` - Delete callback

Uses useSeatRaceSession hook to fetch full session with pieces, boats, assignments.

Component structure:
```tsx
function SessionDetail({ sessionId, onClose, onDelete }) {
  const { data: session, isLoading, error } = useSeatRaceSession(sessionId);
  const deleteSession = useDeleteSession();
  const processSession = useProcessSession();

  if (isLoading) return <LoadingSkeleton />;
  if (error || !session) return <ErrorState message="Session not found" />;

  const handleDelete = async () => {
    if (!confirm('Delete this seat race session?')) return;
    await deleteSession.mutateAsync({ sessionId });
    onClose();
  };

  const handleRecalculate = async () => {
    await processSession.mutateAsync({ sessionId });
  };

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex justify-between items-start">
        <div>
          <h2 className="text-xl font-semibold">{formatDate(session.date)}</h2>
          <div className="flex gap-2 mt-1">
            <Badge>{session.boatClass}</Badge>
            {session.conditions && <Badge variant="outline">{session.conditions}</Badge>}
          </div>
        </div>
        <button onClick={onClose}>
          <XIcon className="w-5 h-5" />
        </button>
      </div>

      {/* Location & Description */}
      {session.location && (
        <div className="text-sm text-muted">
          <MapPinIcon className="w-4 h-4 inline mr-1" />
          {session.location}
        </div>
      )}
      {session.description && (
        <p className="text-sm">{session.description}</p>
      )}

      {/* Pieces */}
      <div className="space-y-4">
        <h3 className="font-medium">Pieces ({session.pieces.length})</h3>
        {session.pieces.map((piece, i) => (
          <PieceDetail key={piece.id} piece={piece} index={i} />
        ))}
      </div>

      {/* Actions */}
      <div className="flex gap-2 pt-4 border-t">
        <button onClick={handleRecalculate} className="btn-secondary">
          Recalculate Ratings
        </button>
        <button onClick={handleDelete} className="btn-danger">
          Delete Session
        </button>
      </div>
    </div>
  );
}
```

**PieceDetail sub-component:**

```tsx
function PieceDetail({ piece, index }) {
  return (
    <div className="border rounded-lg p-4">
      <div className="flex justify-between items-center mb-3">
        <h4 className="font-medium">Piece #{index + 1}</h4>
        {piece.direction && <Badge variant="outline">{piece.direction}</Badge>}
      </div>

      {/* Boats with times */}
      <div className="space-y-2">
        {piece.boats
          .sort((a, b) => (a.finishTimeSeconds || 999) - (b.finishTimeSeconds || 999))
          .map((boat, j) => (
            <BoatDetail key={boat.id} boat={boat} rank={j + 1} />
          ))}
      </div>
    </div>
  );
}
```

```tsx
function BoatDetail({ boat, rank }) {
  return (
    <div className="flex items-center justify-between p-2 bg-surface rounded">
      <div className="flex items-center gap-3">
        <span className={`w-6 text-center font-mono ${rank === 1 ? 'text-green-500 font-bold' : ''}`}>
          {rank}
        </span>
        <span className="font-medium">{boat.name}</span>
      </div>
      <span className="font-mono">
        {boat.finishTimeSeconds ? formatTimeDisplay(boat.finishTimeSeconds) : '—'}
      </span>
    </div>
  );

  {/* Assignments */}
  {boat.assignments && boat.assignments.length > 0 && (
    <div className="mt-2 flex flex-wrap gap-1">
      {boat.assignments
        .sort((a, b) => a.seatNumber - b.seatNumber)
        .map(a => (
          <span key={a.id} className="text-xs bg-surface-elevated px-2 py-1 rounded">
            {a.seatNumber}: {a.athlete?.lastName || 'Unknown'}
          </span>
        ))}
    </div>
  )}
}
```

Create the main Seat Racing page with tabs for Rankings and Sessions.

Component structure:
```tsx
import { useState } from 'react';
import { Tab } from '@headlessui/react';
import { PlusIcon } from 'lucide-react';
import {
  RankingsTable,
  RankingsChart,
  SessionList,
  SessionDetail,
  SessionWizard,
} from '@/v2/components/seat-racing';
import {
  useSeatRaceSessions,
  useAthleteRatings,
  useRecalculateRatings,
} from '@/v2/hooks/useSeatRaceSessions';
import { useAthleteRatings as useRatingsHook } from '@/v2/hooks/useAthleteRatings';

function SeatRacingPage() {
  const [selectedTab, setSelectedTab] = useState(0);
  const [showWizard, setShowWizard] = useState(false);
  const [selectedSessionId, setSelectedSessionId] = useState<string | null>(null);
  const [sideFilter, setSideFilter] = useState<'all' | 'Port' | 'Starboard'>('all');

  // Data fetching
  const { data: sessions, isLoading: sessionsLoading } = useSeatRaceSessions();
  const { data: ratings, isLoading: ratingsLoading } = useRatingsHook({
    side: sideFilter === 'all' ? undefined : sideFilter,
  });
  const recalculateRatings = useRecalculateRatings();

  const handleSessionComplete = (session) => {
    setShowWizard(false);
    setSelectedSessionId(session.id);
  };

  const handleRecalculate = async () => {
    await recalculateRatings.mutateAsync();
  };

  return (
    <div className="p-6 max-w-7xl mx-auto">
      {/* Page Header */}
      <div className="flex justify-between items-center mb-6">
        <div>
          <h1 className="text-2xl font-display font-semibold tracking-tight">
            Seat Racing
          </h1>
          <p className="text-sm text-muted mt-1">
            Compare athletes and track ELO rankings
          </p>
        </div>
        <button
          onClick={() => setShowWizard(true)}
          className="btn-primary flex items-center gap-2"
        >
          <PlusIcon className="w-4 h-4" />
          New Session
        </button>
      </div>

      {/* Tabs */}
      <Tab.Group selectedIndex={selectedTab} onChange={setSelectedTab}>
        <Tab.List className="flex gap-4 border-b border-border mb-6">
          <Tab className={({ selected }) =>
            `px-4 py-2 -mb-px border-b-2 transition-colors ${
              selected
                ? 'border-primary text-primary'
                : 'border-transparent text-muted hover:text-primary'
            }`
          }>
            Rankings
          </Tab>
          <Tab className={({ selected }) =>
            `px-4 py-2 -mb-px border-b-2 transition-colors ${
              selected
                ? 'border-primary text-primary'
                : 'border-transparent text-muted hover:text-primary'
            }`
          }>
            Sessions
          </Tab>
        </Tab.List>

        <Tab.Panels>
          {/* Rankings Tab */}
          <Tab.Panel>
            <div className="space-y-6">
              {/* Side filter */}
              <div className="flex gap-2">
                {['all', 'Port', 'Starboard'].map((side) => (
                  <button
                    key={side}
                    onClick={() => setSideFilter(side as typeof sideFilter)}
                    className={`px-3 py-1 rounded text-sm ${
                      sideFilter === side
                        ? 'bg-primary text-primary-foreground'
                        : 'bg-surface hover:bg-surface-elevated'
                    }`}
                  >
                    {side === 'all' ? 'All Sides' : side}
                  </button>
                ))}
              </div>

              {/* Rankings visualization */}
              <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
                <div className="bg-surface rounded-lg p-4">
                  <h3 className="font-medium mb-4">ELO Distribution</h3>
                  <RankingsChart ratings={ratings || []} />
                </div>
                <div className="bg-surface rounded-lg p-4">
                  <RankingsTable
                    ratings={ratings || []}
                    isLoading={ratingsLoading}
                    onRecalculate={handleRecalculate}
                  />
                </div>
              </div>
            </div>
          </Tab.Panel>

          {/* Sessions Tab */}
          <Tab.Panel>
            <SessionList
              sessions={sessions?.sessions || []}
              isLoading={sessionsLoading}
              onSelectSession={setSelectedSessionId}
              onDeleteSession={(id) => {
                if (selectedSessionId === id) setSelectedSessionId(null);
              }}
            />
          </Tab.Panel>
        </Tab.Panels>
      </Tab.Group>

      {/* Session Wizard Modal */}
      {showWizard && (
        <Modal onClose={() => setShowWizard(false)} size="xl">
          <SessionWizard
            onComplete={handleSessionComplete}
            onCancel={() => setShowWizard(false)}
          />
        </Modal>
      )}

      {/* Session Detail Slide-out */}
      {selectedSessionId && (
        <SlideOutPanel onClose={() => setSelectedSessionId(null)}>
          <SessionDetail
            sessionId={selectedSessionId}
            onClose={() => setSelectedSessionId(null)}
          />
        </SlideOutPanel>
      )}
    </div>
  );
}

export default SeatRacingPage;
```

Use existing Modal and SlideOutPanel components from V2 shared components, or create simple versions inline:

```tsx
function Modal({ children, onClose, size = 'md' }) {
  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center">
      <div className="absolute inset-0 bg-black/50" onClick={onClose} />
      <div className={`relative bg-elevated rounded-lg p-6 max-h-[90vh] overflow-auto ${
        size === 'xl' ? 'w-full max-w-4xl' : 'w-full max-w-lg'
      }`}>
        {children}
      </div>
    </div>
  );
}

function SlideOutPanel({ children, onClose }) {
  return (
    <div className="fixed inset-0 z-50 flex justify-end">
      <div className="absolute inset-0 bg-black/30" onClick={onClose} />
      <div className="relative w-full max-w-lg bg-elevated h-full overflow-auto p-6">
        {children}
      </div>
    </div>
  );
}
```
