function AthleteAssignmentStep() {
  const { watch, setValue } = useFormContext();
  const { data: athletes, isLoading } = useAthletes();
  const pieces = watch('pieces');
  const boatClass = watch('boatClass');

  // Determine seats needed based on boat class
  const seatConfig = getBoatSeatConfig(boatClass);
  // e.g., '8+' => { seats: 8, hasCox: true }
  // '4-' => { seats: 4, hasCox: false }

  // Track all assigned athletes across all boats
  const assignedAthleteIds = useMemo(() => {
    const ids = new Set<string>();
    pieces.forEach(piece => {
      piece.boats.forEach(boat => {
        boat.assignments?.forEach(a => ids.add(a.athleteId));
      });
    });
    return ids;
  }, [pieces]);

  return (
    <div className="space-y-8">
      {pieces.map((piece, pieceIndex) => (
        <div key={pieceIndex} className="border rounded-lg p-4">
          <h3>Piece #{piece.sequenceOrder}</h3>

          <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mt-4">
            {piece.boats.map((boat, boatIndex) => (
              <BoatAssignmentCard
                key={boatIndex}
                pieceIndex={pieceIndex}
                boatIndex={boatIndex}
                boat={boat}
                seatConfig={seatConfig}
                athletes={athletes}
                assignedAthleteIds={assignedAthleteIds}
              />

  );
}

**BoatAssignmentCard sub-component:**
```tsx
function BoatAssignmentCard({ pieceIndex, boatIndex, boat, seatConfig, athletes, assignedAthleteIds }) {
  const { setValue, watch } = useFormContext();
  const fieldPrefix = `pieces.${pieceIndex}.boats.${boatIndex}`;
  const assignments = watch(`${fieldPrefix}.assignments`) || [];

  // Generate seat slots based on boat class
  const seatSlots = generateSeatSlots(seatConfig);
  // e.g., [{ seatNumber: 1, side: 'Starboard', label: 'Bow' }, ...]

  const handleAssign = (seatNumber: number, side: string, athleteId: string | null) => {
    const newAssignments = [...assignments];
    const existingIndex = newAssignments.findIndex(a => a.seatNumber === seatNumber);

    if (athleteId === null) {
      // Remove assignment
      if (existingIndex >= 0) newAssignments.splice(existingIndex, 1);
    } else {
      const assignment = { athleteId, seatNumber, side };
      if (existingIndex >= 0) {
        newAssignments[existingIndex] = assignment;
      } else {
        newAssignments.push(assignment);
      }
    }

    setValue(`${fieldPrefix}.assignments`, newAssignments);
  };

  return (
    <div className="bg-surface rounded-lg p-4">
      <h4 className="font-medium mb-3">{boat.name}</h4>
      <div className="space-y-2">
        {seatSlots.map(slot => {
          const assignment = assignments.find(a => a.seatNumber === slot.seatNumber);
          return (
            <div key={slot.seatNumber} className="flex items-center gap-3">
              <span className="w-16 text-sm text-muted">{slot.label}</span>
              <SeatSlotSelector
                value={assignment?.athleteId || null}
                onChange={(id) => handleAssign(slot.seatNumber, slot.side, id)}
                athletes={athletes}
                seatNumber={slot.seatNumber}
                side={slot.side}
                disabledAthleteIds={[...assignedAthleteIds].filter(id => id !== assignment?.athleteId)}
              />
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

Helper function `generateSeatSlots(config)`:
```typescript
function generateSeatSlots(config: { seats: number; hasCox: boolean }) {
  const slots = [];
  // Sweep boats alternate port/starboard
  for (let i = 1; i <= config.seats; i++) {
    slots.push({
      seatNumber: i,
      side: i % 2 === 0 ? 'Port' : 'Starboard',
      label: i === 1 ? 'Bow' : i === config.seats ? 'Stroke' : `Seat ${i}`,
    });
  }
  if (config.hasCox) {
    slots.push({ seatNumber: config.seats + 1, side: 'Cox', label: 'Cox' });
  }
  return slots;
}
```

Helper function `getBoatSeatConfig(boatClass)`:
```typescript
function getBoatSeatConfig(boatClass: string) {
  const configs = {
    '8+': { seats: 8, hasCox: true },
    '4+': { seats: 4, hasCox: true },
    '4-': { seats: 4, hasCox: false },
    '4x': { seats: 4, hasCox: false },
    '2-': { seats: 2, hasCox: false },
    '2x': { seats: 2, hasCox: false },
    '1x': { seats: 1, hasCox: false },
  };
  return configs[boatClass] || { seats: 4, hasCox: true };
}
```
