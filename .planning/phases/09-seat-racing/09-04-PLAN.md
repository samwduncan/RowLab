```typescript
function parseTimeInput(input: string): number | null {
  if (!input || input.trim() === '') return null;

  // Try MM:SS.s or MM:SS format
  const colonMatch = input.match(/^(\d+):(\d{1,2}(?:\.\d+)?)$/);
  if (colonMatch) {
    const mins = parseInt(colonMatch[1], 10);
    const secs = parseFloat(colonMatch[2]);
    return mins * 60 + secs;
  }

  // Try plain seconds
  const num = parseFloat(input);
  if (!isNaN(num) && num > 0) return num;

  return null;
}

function formatTimeDisplay(seconds: number | null): string {
  if (seconds === null) return '';
  const mins = Math.floor(seconds / 60);
  const secs = (seconds % 60).toFixed(1);
  return `${mins}:${secs.padStart(4, '0')}`;
}
```

- On focus: show raw input
- On blur: parse and format to MM:SS.s

- V2 input tokens (bg-surface, border-border, focus:ring-primary)

```tsx
function PieceManagerStep() {
  const { control, register, formState: { errors } } = useFormContext();
  const { fields: pieces, append: addPiece, remove: removePiece } = useFieldArray({
    control,
    name: 'pieces',
  });

  const handleAddPiece = () => {
    addPiece({
      sequenceOrder: pieces.length + 1,
      distanceMeters: 500,
      direction: null,
      notes: '',
      boats: [
        { name: 'Boat A', finishTimeSeconds: null, handicapSeconds: 0 },
        { name: 'Boat B', finishTimeSeconds: null, handicapSeconds: 0 },
      ],
    });
  };

  return (
    <div className="space-y-6">
      {/* Add Piece button */}
      <div className="flex justify-between items-center">
        <h3>Pieces</h3>
        <button type="button" onClick={handleAddPiece}>+ Add Piece</button>
      </div>

      {/* Piece list */}
      {pieces.map((piece, pieceIndex) => (
        <PieceCard
          key={piece.id}
          pieceIndex={pieceIndex}
          onRemove={() => removePiece(pieceIndex)}
        />
      ))}

      {/* Empty state */}
      {pieces.length === 0 && (
        <EmptyState message="Add at least one piece to continue" />
      )}
    </div>
  );
}
```

```tsx
function BoatRow({ pieceIndex, boatIndex, onRemove }) {
  const { register, setValue, watch } = useFormContext();
  const fieldPrefix = `pieces.${pieceIndex}.boats.${boatIndex}`;
  const timeSeconds = watch(`${fieldPrefix}.finishTimeSeconds`);

  return (
    <div className="flex items-center gap-3">
      <input
        {...register(`${fieldPrefix}.name`)}
        placeholder="Boat name"
        className="w-32"
      />
      <BoatTimeEntry
        value={formatTimeDisplay(timeSeconds)}
        onChange={(secs) => setValue(`${fieldPrefix}.finishTimeSeconds`, secs)}
      />
      <button type="button" onClick={onRemove}>
        <TrashIcon className="w-4 h-4" />
      </button>
    </div>
  );
}
```
