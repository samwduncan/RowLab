Create a new service for NCAA compliance calculations that:

1. **getWeeklyCompliance(teamId, weekStart, athleteId?)**: Returns NCAAAuditEntry[] for the week
   - Queries PracticeSession or derives from WorkoutCompletion + Attendance
   - Calculates daily hours per athlete
   - Flags isNearLimit (>=18h) and isOverLimit (>20h)
   - If athleteId provided, filter to single athlete

2. **getComplianceReport(teamId, weekStart)**: Returns NCAAComplianceReport
   - Aggregates weekly compliance into audit report format
   - Includes all athlete entries with sessions
   - Counts athletesNearLimit and athletesOverLimit
   - Includes generatedAt timestamp

3. **getTrainingLoad(teamId, startDate, endDate, athleteId?)**: Returns WeeklyLoad[]
   - Groups WorkoutCompletions by week
   - Calculates TSS and volume (minutes) per week
   - Returns array of week summaries

4. **linkAttendanceToTraining(teamId, date)**: Links attendance records to scheduled training sessions (ATT-04)
   - Finds scheduled PlannedWorkouts for the date
   - Maps athlete attendance status to training session participation
   - Returns linked records for compliance tracking

Implementation pattern:
```javascript
// server/services/ncaaComplianceService.js

import prisma from '../db/prisma.js';
import { startOfWeek, endOfWeek, format, addDays, eachWeekOfInterval, parseISO } from 'date-fns';
import logger from '../utils/logger.js';

// NCAA limits
const WEEKLY_LIMIT_HOURS = 20;
const DAILY_LIMIT_HOURS = 4;
const WARNING_THRESHOLD_HOURS = 18;
const COMPETITION_HOURS = 3;

/**
 * Get weekly NCAA compliance data for team
 */
export async function getWeeklyCompliance(teamId, weekStartDate, athleteId = null) {
  const weekStart = startOfWeek(new Date(weekStartDate), { weekStartsOn: 1 });
  const weekEnd = endOfWeek(weekStart, { weekStartsOn: 1 });

  // Get all athletes on the team (or filter to one)
  const whereClause = { teamId };
  if (athleteId) whereClause.id = athleteId;

  const athletes = await prisma.athlete.findMany({
    where: whereClause,
    select: { id: true, name: true },
  });

  // Get workout completions in date range
  const completions = await prisma.workoutCompletion.findMany({
    where: {
      athleteId: athleteId ? athleteId : { in: athletes.map(a => a.id) },
      completedAt: {
        gte: weekStart,
        lte: weekEnd,
      },
    },
    include: {
      plannedWorkout: true,
    },
  });

  // Get attendance records to link to training sessions (ATT-04)
  const attendanceRecords = await prisma.attendance.findMany({
    where: {
      teamId,
      date: {
        gte: format(weekStart, 'yyyy-MM-dd'),
        lte: format(weekEnd, 'yyyy-MM-dd'),
      },
      ...(athleteId && { athleteId }),
    },
  });

  // Build compliance entries per athlete
  const entries = athletes.map(athlete => {
    const athleteCompletions = completions.filter(c => c.athleteId === athlete.id);
    const athleteAttendance = attendanceRecords.filter(a => a.athleteId === athlete.id);

    // Calculate daily hours from completions
    const dailyHours = {};
    for (let d = 0; d < 7; d++) {
      const day = addDays(weekStart, d);
      const dateKey = format(day, 'yyyy-MM-dd');
      dailyHours[dateKey] = 0;
    }

    // Add hours from workout completions
    for (const completion of athleteCompletions) {
      const dateKey = format(new Date(completion.completedAt), 'yyyy-MM-dd');
      const durationHours = (completion.plannedWorkout?.duration || 0) / 3600; // seconds to hours
      if (dailyHours[dateKey] !== undefined) {
        dailyHours[dateKey] += durationHours;
      }
    }

    // Add hours from attendance records that aren't linked to completions (ATT-04)
    // This handles practice attendance that wasn't a specific workout
    for (const attendance of athleteAttendance) {
      if (attendance.status === 'present' || attendance.status === 'late') {
        // If attendance has duration info from a practice session, add it
        // Default 2 hours for practice attendance without specific duration
        const durationHours = attendance.durationMinutes ? attendance.durationMinutes / 60 : 2;
        const dateKey = attendance.date;

        // Only add if not already counted via completion
        const hasCompletionForDay = athleteCompletions.some(c =>
          format(new Date(c.completedAt), 'yyyy-MM-dd') === dateKey
        );

        if (!hasCompletionForDay && dailyHours[dateKey] !== undefined) {
          dailyHours[dateKey] += durationHours;
        }
      }
    }

    // Round daily hours
    for (const key of Object.keys(dailyHours)) {
      dailyHours[key] = Math.round(dailyHours[key] * 100) / 100;
    }

    const weeklyTotal = Object.values(dailyHours).reduce((sum, h) => sum + h, 0);

    return {
      athleteId: athlete.id,
      athleteName: athlete.name,
      weekStart: format(weekStart, 'yyyy-MM-dd'),
      dailyHours,
      weeklyTotal: Math.round(weeklyTotal * 100) / 100,
      isNearLimit: weeklyTotal >= WARNING_THRESHOLD_HOURS,
      isOverLimit: weeklyTotal > WEEKLY_LIMIT_HOURS,
      sessions: [], // Could populate with actual session records if needed
    };
  });

  return entries;
}

/**
 * Get full compliance report for audit purposes
 */
export async function getComplianceReport(teamId, weekStartDate) {
  const weekStart = startOfWeek(new Date(weekStartDate), { weekStartsOn: 1 });
  const weekEnd = endOfWeek(weekStart, { weekStartsOn: 1 });

  const entries = await getWeeklyCompliance(teamId, weekStartDate);

  return {
    weekStart: format(weekStart, 'yyyy-MM-dd'),
    weekEnd: format(weekEnd, 'yyyy-MM-dd'),
    teamId,
    entries,
    athletesNearLimit: entries.filter(e => e.isNearLimit && !e.isOverLimit).length,
    athletesOverLimit: entries.filter(e => e.isOverLimit).length,
    generatedAt: new Date().toISOString(),
  };
}

/**
 * Get training load (TSS/volume) over date range
 */
export async function getTrainingLoad(teamId, startDate, endDate, athleteId = null) {
  const start = new Date(startDate);
  const end = new Date(endDate);

  // Get completions in range
  const whereClause = {
    completedAt: {
      gte: start,
      lte: end,
    },
  };

  if (athleteId) {
    whereClause.athleteId = athleteId;
  } else {
    // Filter to team's athletes
    const teamAthletes = await prisma.athlete.findMany({
      where: { teamId },
      select: { id: true },
    });
    whereClause.athleteId = { in: teamAthletes.map(a => a.id) };
  }

  const completions = await prisma.workoutCompletion.findMany({
    where: whereClause,
    include: {
      plannedWorkout: true,
    },
  });

  // Group by week
  const weeks = eachWeekOfInterval({ start, end }, { weekStartsOn: 1 });
  const weeklyData = weeks.map(weekStart => {
    const weekEnd = endOfWeek(weekStart, { weekStartsOn: 1 });
    const weekCompletions = completions.filter(c => {
      const completedDate = new Date(c.completedAt);
      return completedDate >= weekStart && completedDate <= weekEnd;
    });

    const tss = weekCompletions.reduce((sum, c) => sum + (c.plannedWorkout?.estimatedTSS || 0), 0);
    const volume = weekCompletions.reduce((sum, c) => sum + (c.plannedWorkout?.duration || 0) / 60, 0); // minutes

    return {
      week: format(weekStart, 'MMM d'),
      weekStart: format(weekStart, 'yyyy-MM-dd'),
      tss,
      volume: Math.round(volume),
      workoutCount: weekCompletions.length,
    };
  });

  return weeklyData;
}

/**
 * Link attendance to training sessions for a specific date (ATT-04)
 * This function creates the connection between attendance records and scheduled training workouts
 */
export async function linkAttendanceToTraining(teamId, date) {
  const dateStr = format(new Date(date), 'yyyy-MM-dd');

  // Get attendance for the date
  const attendance = await prisma.attendance.findMany({
    where: {
      teamId,
      date: dateStr,
    },
    include: {
      athlete: true,
    },
  });

  // Get scheduled workouts for that date (from all active assignments)
  const scheduledWorkouts = await prisma.plannedWorkout.findMany({
    where: {
      scheduledDate: {
        gte: new Date(dateStr),
        lt: new Date(dateStr + 'T23:59:59'),
      },
      plan: {
        teamId,
        isTemplate: false,
      },
    },
    include: {
      plan: true,
    },
  });

  // Create linkage - which athletes were present for which workouts
  const linkedRecords = attendance.map(att => {
    const workoutsForDate = scheduledWorkouts.filter(w => {
      // Check if this athlete is assigned to this plan
      // For now, assume all team workouts apply to all present athletes
      return true;
    });

    return {
      athleteId: att.athleteId,
      athleteName: att.athlete?.name,
      date: dateStr,
      attendanceStatus: att.status,
      scheduledWorkouts: workoutsForDate.map(w => ({
        id: w.id,
        name: w.name,
        type: w.type,
        duration: w.duration,
        planName: w.plan?.name,
      })),
      participated: att.status === 'present' || att.status === 'late',
    };
  });

  return linkedRecords;
}

export default {
  getWeeklyCompliance,
  getComplianceReport,
  getTrainingLoad,
  linkAttendanceToTraining,
};
```

```javascript
// ============================================
// NCAA COMPLIANCE

/**
 * GET /api/v1/training-plans/compliance/weekly
 * Get weekly NCAA compliance data for team
 */
router.get(
  '/compliance/weekly',
  authenticateToken,
  teamIsolation,
  [
    query('weekStart').isISO8601(),
    query('athleteId').optional().isUUID(),
  ],
  validateRequest,
  async (req, res) => {
    try {
      const entries = await getWeeklyCompliance(
        req.user.activeTeamId,
        req.query.weekStart,
        req.query.athleteId
      );
      res.json({
        success: true,
        data: { entries },
      });
    } catch (error) {
      logger.error('Get weekly compliance error', { error: error.message });
      res.status(500).json({
        success: false,
        error: { code: 'SERVER_ERROR', message: 'Failed to get compliance data' },
      });
    }
  }
);
```
