Create a toggle button component that allows users to switch between V2 and V1:

```typescript
// src/v2/components/shell/VersionToggle.tsx
import { useNavigate } from 'react-router-dom';
import { useUserPreferenceStore } from '@v2/stores/userPreferenceStore';

interface VersionToggleProps {
  /** Which version this toggle is rendered in */
  currentVersion: 'v1' | 'v2';
}

/**
 * Toggle button to switch between V1 (legacy) and V2 versions.
 * When clicked, updates preference and redirects to the other version.
 */
export function VersionToggle({ currentVersion }: VersionToggleProps) {
  const setLegacyMode = useUserPreferenceStore((state) => state.setLegacyMode);
  const navigate = useNavigate();

  const handleToggle = () => {
    if (currentVersion === 'v2') {
      // Currently on V2, switch to legacy
      setLegacyMode(true);
      navigate('/legacy', { replace: true });
    } else {
      // Currently on V1 legacy, switch to V2
      setLegacyMode(false);
      navigate('/app', { replace: true });
    }
  };

  if (currentVersion === 'v2') {
    return (
      <button
        onClick={handleToggle}
        className="px-3 py-1.5 text-sm text-txt-secondary hover:text-txt-primary hover:bg-bg-surface-elevated rounded-md transition-colors"
        title="Switch to legacy version"
      >
        Use Legacy
      </button>
    );
  }

  // V1 version - use V1 styling (no V2 tokens)
  return (
    <button
      onClick={handleToggle}
      className="px-3 py-1.5 text-sm text-text-secondary hover:text-text-primary hover:bg-dark-card rounded-md transition-colors"
      title="Try the new RowLab"
    >
      Try New Version
    </button>
  );
}
```

Key behaviors:
- Different styling for V1 vs V2 contexts (V1 uses old tokens, V2 uses new tokens)
- Updates preference store immediately
- Navigates with replace: true to avoid history pollution
- Clear labels: "Use Legacy" in V2, "Try New Version" in V1

1. Create VersionRedirectGuard component:
```typescript
// src/v2/components/shell/VersionRedirectGuard.tsx
import { useVersionRedirect } from '@v2/hooks/useVersionRedirect';

/**
 * Guard component that calls useVersionRedirect hook to handle
 * automatic redirects based on user version preference.
 *
 * Must be rendered inside Router context (inside V2Layout).
 */
export function VersionRedirectGuard({ children }: { children: React.ReactNode }) {
  // This hook watches useLegacyMode preference and redirects if needed
  useVersionRedirect();

  return <>{children}</>;
}
```

2. Update V2Layout.tsx to use VersionRedirectGuard:

Import the guard:
```typescript
import { VersionRedirectGuard } from '@v2/components/shell/VersionRedirectGuard';
```

Wrap the layout content with VersionRedirectGuard (inside QueryClientProvider and Context providers, but wrapping the main content):

```tsx
// Inside V2Layout return statement, wrap the main content area
<VersionRedirectGuard>
  <Outlet />
</VersionRedirectGuard>
```

**Why a wrapper component instead of calling hook directly in V2Layout?**
- V2Layout renders providers that useVersionRedirect depends on
- Separating the hook call into a child component ensures all providers are available
- Makes testing easier (can test redirect logic independently)
- Follows composition pattern

**Result:** When a user with useLegacyMode=true loads any V2 page, they will be automatically redirected to /legacy based on the preference stored in localStorage.
