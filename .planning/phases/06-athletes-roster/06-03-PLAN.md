export function useAthletes(filters?: AthleteFilters) {
  const queryClient = useQueryClient();
  const isAuthenticated = useAuthStore((state) => state.isAuthenticated);
  const isInitialized = useAuthStore((state) => state.isInitialized);

  // Main query with filter support
  const query = useQuery({
    queryKey: ['athletes', filters],
    queryFn: () => fetchAthletes(filters),
    enabled: isInitialized && isAuthenticated,
    staleTime: 2 * 60 * 1000, // 2 minutes
  });

  // Create mutation with optimistic update
  const createMutation = useMutation({...});

  // Update mutation with optimistic update
  const updateMutation = useMutation({...});

  // Delete mutation with optimistic update
  const deleteMutation = useMutation({...});

  // Bulk import mutation (no optimistic update)
  const bulkImportMutation = useMutation({...});

  return {
    athletes: query.data || [],
    isLoading: query.isLoading,

    refetch: query.refetch,

// Team attendance for a date (coach view)
export function useAttendance(date?: string) {
  const queryClient = useQueryClient();
  const isAuthenticated = useAuthStore((state) => state.isAuthenticated);
  const isInitialized = useAuthStore((state) => state.isInitialized);

  const query = useQuery({
    queryKey: ['attendance', 'team', date],
    queryFn: () => fetchTeamAttendance(date),
    enabled: isInitialized && isAuthenticated && !!date,
    staleTime: 30 * 1000, // 30 seconds - attendance changes during practice
  });

  const recordMutation = useMutation({
    mutationFn: (data: { date: string; records: AttendanceInput[] }) =>
      recordAttendance(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['attendance'] });
    },
  });

  return {
    attendance: query.data || [],
    isLoading: query.isLoading,
    error: query.error,
    refetch: query.refetch,

    recordAttendance: recordMutation.mutate,
    isRecording: recordMutation.isPending,
    recordError: recordMutation.error,
  };
}

// Individual athlete attendance history
export function useAthleteAttendance(
  athleteId: string,
  options?: { startDate?: string; endDate?: string; limit?: number }
) {
  const isAuthenticated = useAuthStore((state) => state.isAuthenticated);
  const isInitialized = useAuthStore((state) => state.isInitialized);

  return useQuery({
    queryKey: ['attendance', 'athlete', athleteId, options],
    queryFn: () => fetchAthleteAttendance(athleteId, options),
    enabled: isInitialized && isAuthenticated && !!athleteId,
    staleTime: 60 * 1000, // 1 minute
  });
}
