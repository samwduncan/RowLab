videos.forEach((v, i) => {
  // Create placeholder frame
  const frame = penpot.createFrame({
    name: `VIDEO: ${v.name}`,
    x: 0,
    y: 1500 + i * 200,
    width: v.width / 3, // Scaled down for reference
    height: v.height / 3
  });

  // Add text label "VIDEO" prominently
  penpot.createText({
    content: 'VIDEO',
    x: frame.x + 10,
    y: frame.y + 10,
    fontSize: 24,
    fontWeight: 700
  });

  // Add usage note
  penpot.createText({
    content: `${v.name}\n${v.usage}`,
    x: frame.x + 10,
    y: frame.y + 50,
    fontSize: 12
  });
});
