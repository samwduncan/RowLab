```javascript
// Use penpot.library.local.createTypography() for each style
const typographyStyles = [
  // Display (Fraunces - editorial serif)
  { name: 'Display/Hero', fontFamily: 'Fraunces', fontSize: 72, fontWeight: 600, lineHeight: 1.1 },
  { name: 'Display/H1', fontFamily: 'Fraunces', fontSize: 48, fontWeight: 600, lineHeight: 1.15 },
  { name: 'Display/H2', fontFamily: 'Fraunces', fontSize: 36, fontWeight: 600, lineHeight: 1.2 },
  { name: 'Display/H3', fontFamily: 'Fraunces', fontSize: 24, fontWeight: 600, lineHeight: 1.25 },

  // Body (Inter - clean sans)
  { name: 'Body/Large', fontFamily: 'Inter', fontSize: 18, fontWeight: 400, lineHeight: 1.75 },
  { name: 'Body/Base', fontFamily: 'Inter', fontSize: 15, fontWeight: 400, lineHeight: 1.5 },
  { name: 'Body/Small', fontFamily: 'Inter', fontSize: 13, fontWeight: 400, lineHeight: 1.5 },
  { name: 'Body/XS', fontFamily: 'Inter', fontSize: 12, fontWeight: 400, lineHeight: 1.33 },

  // Metric (Geist Mono - data/numbers)
  { name: 'Metric/Hero', fontFamily: 'Geist Mono', fontSize: 72, fontWeight: 600, lineHeight: 1.1 },
  { name: 'Metric/Large', fontFamily: 'Geist Mono', fontSize: 48, fontWeight: 600, lineHeight: 1.15 },
  { name: 'Metric/Medium', fontFamily: 'Geist Mono', fontSize: 24, fontWeight: 500, lineHeight: 1.25 },
  { name: 'Metric/Small', fontFamily: 'Geist Mono', fontSize: 13, fontWeight: 500, lineHeight: 1.25 },

  // Label (Inter - UI labels)
  { name: 'Label/Large', fontFamily: 'Inter', fontSize: 14, fontWeight: 500, lineHeight: 1.5 },
  { name: 'Label/Default', fontFamily: 'Inter', fontSize: 13, fontWeight: 500, lineHeight: 1.5 },
  { name: 'Label/Small', fontFamily: 'Inter', fontSize: 11, fontWeight: 500, lineHeight: 1.33 }
];

typographyStyles.forEach(t => {
  penpot.library.local.createTypography(t.name, {
    fontFamily: t.fontFamily,
    fontSize: t.fontSize,
    fontWeight: t.fontWeight,
    lineHeight: t.lineHeight
  });
});
```

Note: If Geist Mono is unavailable in Penpot, use JetBrains Mono as fallback and document the substitution.
